            if (d) {
               Throwable e;
               if (this.delayError) {
                  if (empty) {
                     e = this.error;
                     if (e != null) {
                        this.doError(a, e);
                     } else {
                        this.doComplete(a);
                     }

                     return true;
                  }
               } else {
                  e = this.error;
                  if (e != null) {
                     Operators.onDiscard(v, this.actual.currentContext());
                     if (this.sourceMode == 2) {
                        this.queue.clear();
                     } else {
                        Operators.onDiscardQueueWithClear(this.queue, this.actual.currentContext(), (Function)null);
                     }

                     this.doError(a, e);
                     return true;
                  }

                  if (empty) {
                     this.doComplete(a);
                     return true;
                  }
               }
            }

            return false;
         }
      }

      @Nullable
      public Object scanUnsafe(Attr key) {
         if (key == Attr.REQUESTED_FROM_DOWNSTREAM) {
            return this.requested;
         } else if (key == Attr.PARENT) {
            return this.s;
         } else if (key == Attr.CANCELLED) {
            return this.cancelled;
         } else if (key == Attr.TERMINATED) {
            return this.done;
         } else if (key == Attr.BUFFERED) {
            return this.queue != null ? this.queue.size() : 0;
         } else if (key == Attr.ERROR) {
            return this.error;
         } else if (key == Attr.DELAY_ERROR) {
            return this.delayError;
         } else if (key == Attr.PREFETCH) {
            return this.prefetch;
         } else {
            return key == Attr.RUN_ON ? this.worker : super.scanUnsafe(key);
         }
      }

         return this.actual;

      public void clear() {
         if (DISCARD_GUARD.getAndIncrement(this) == 0) {
            int missed = 1;

            do {
               while(true) {
                  Operators.onDiscardQueueWithClear(this.queue, this.actual.currentContext(), (Function)null);
                  int dg = this.discardGuard;
                  if (missed == dg) {
                     missed = DISCARD_GUARD.addAndGet(this, -missed);
                     break;
                  }

                  missed = dg;
               }
            } while(missed != 0);

         }
      }

         return this.queue.isEmpty();

      @Nullable
      public T poll() {
         T v = this.queue.poll();
         if (v != null && this.sourceMode != 1) {
            long p = this.produced + 1L;
            if (p == (long)this.limit) {
               this.produced = 0L;
               this.s.request(p);
            } else {
               this.produced = p;
            }
         }

         return v;
      }
